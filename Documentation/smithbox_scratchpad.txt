    private unsafe void SetupFonts()
    {
        var style = ImGui.GetStyle();
        style.FontSizeBase = CFG.Current.Interface_Font_Size;

        var io = ImGui.GetIO();

        ImFontAtlasPtr fonts = ImGui.GetIO().Fonts;
        fonts.Clear();

        var iconRange = new uint[] { 0xf000, 0xf339 };

        ImFontConfigPtr cfg_English = ImGui.ImFontConfig();
        cfg_English.MergeMode = true;

        unsafe
        {
            fixed (uint* ptr = iconRange)
            {
                cfg_English.GlyphExcludeRanges = ptr;
            }
        }

        var englishFontPath = Path.Combine(AppContext.BaseDirectory, EnglishFontPath);
        io.Fonts.AddFontFromFileTTF(englishFontPath, style.FontSizeBase, cfg_English);

        ImFontConfigPtr cfg_NonEnglish = ImGui.ImFontConfig();
        cfg_NonEnglish.MergeMode = true;

        unsafe
        {
            fixed (uint* ptr = iconRange)
            {
                cfg_NonEnglish.GlyphExcludeRanges = ptr;
            }
        }

        var nonEnglishFontPath = Path.Combine(AppContext.BaseDirectory, NonEnglishFontPath);
        io.Fonts.AddFontFromFileTTF(nonEnglishFontPath, style.FontSizeBase, cfg_NonEnglish);

        ImFontConfigPtr cfg_Icon = ImGui.ImFontConfig();
        cfg_Icon.MergeMode = true;
        cfg_Icon.GlyphMinAdvanceX = style.FontSizeBase;

        var iconFontPath = Path.Combine(AppContext.BaseDirectory, IconFontPath);
        io.Fonts.AddFontFromFileTTF(iconFontPath, style.FontSizeBase, cfg_Icon);

        _context.ImguiRenderer.RecreateFontDeviceTexture();
    }
